1. Programs and total number of students in each
SELECT p.Program_Name, COUNT(s.Student_ID) AS Total_Students
FROM Program p
JOIN Student s ON p.Program_ID = s.Program_ID
GROUP BY p.Program_Name;

2. Announcements for a particular module between 1st May and 28th May
SELECT *
FROM Announcement a
JOIN Announcement_Module_Teacher amt ON a.Announcement_ID = amt.Announcement_ID
WHERE amt.Module_ID = 201 AND a.Announcement_Date BETWEEN TO_DATE('2024-05-01', 'YYYY-MM-DD') AND TO_DATE('2024-05-28', 'YYYY-MM-DD');

3. Modules starting with 'D' and total uploaded resources
SELECT m.Module_Name, COUNT(rms.Resource_ID) AS Total_Resources
FROM Module m
JOIN Resource_Module_Student rms ON m.Module_ID = rms.Module_ID
WHERE m.Module_Name LIKE 'D%'
GROUP BY m.Module_Name;

4. Students with no assessment submission for a module
SELECT s.Student_Name, p.Program_Name
FROM Student s
JOIN Program p ON s.Program_ID = p.Program_ID
WHERE s.Student_ID NOT IN (
    SELECT DISTINCT Student_ID FROM Module_Student_Assessment WHERE Module_ID = 201
);

5. Teachers teaching more than one module
SELECT Teacher_Name
FROM Teacher t
JOIN Student_Teacher_Module stm ON t.Teacher_ID = stm.Teacher_ID
GROUP BY Teacher_Name
HAVING COUNT(DISTINCT Module_ID) > 1;
